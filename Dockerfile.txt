# استخدم صورة .NET 8.0 SDK لإنشاء المشروع
FROM mcr.microsoft.com/dotnet/sdk:8.0 AS build
WORKDIR /app

# نسخ ملفات المشروع إلى الحاوية
COPY . ./

# استعادة الحزم (Fix لخطأ dotnet restore)
RUN dotnet restore

# بناء المشروع
RUN dotnet publish -c Release -o /app/publish

# استخدم صورة ASP.NET 8.0 لتشغيل التطبيق
FROM mcr.microsoft.com/dotnet/aspnet:8.0 AS runtime
WORKDIR /app
COPY --from=build /app/publish .

# فتح المنفذ 80
EXPOSE 80

# تشغيل التطبيق
CMD ["dotnet", "fitapp.dll"]
